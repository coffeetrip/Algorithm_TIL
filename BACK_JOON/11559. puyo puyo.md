def down(arr):
    for j in range(6):
        for i in range(12 - 1, -1, -1):
            if arr[i][j] == '.':
                for k in range(i - 1, -1, -1):
                    if arr[k][j] != '.':
                        arr[i][j] = arr[k][j]
                        arr[k][j] = '.'
                        break
    return arr


def check(si, sj, t):
    global visited
    q = [[si, sj]]
    visited[si][sj] = t

    while q:
        i, j = q.pop(0)
        for di, dj in (0, 1), (1, 0), (0, -1), (-1, 0):
            ni, nj = i + di, j + dj
            if 0 <= ni < 12 and 0 <= nj < 6 and visited[ni][nj] == 0 and puyo[ni][nj] == puyo[i][j]:
                visited[ni][nj] = t
                q.append([ni, nj])

    cnt = 0
    for i in range(12):
        for j in range(6):
            if visited[i][j] == t:
                cnt += 1

    if cnt >= 4:
        return 1
    else:
        return 0


puyo = [list(input()) for _ in range(12)]
cnt, result, visited_num = 0, 0, 0
while True:
    stop = 0
    visited = [[0] * 6 for _ in range(12)]
    for i in range(12):
        for j in range(6):
            if puyo[i][j] != '.' and visited[i][j] == 0:
                visited_num += 1
                result = check(i, j, visited_num)
                if result == 1:
                    stop += 1
                    for r in range(12):
                        for c in range(6):
                            if visited[r][c] == visited_num:
                                puyo[r][c] = '.'
                    down(puyo)
    if stop == 0:
        break
    cnt += 1

print(cnt)
