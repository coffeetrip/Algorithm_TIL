```python
def f(n, N, r, bene):  # 현재 날짜 순서, 총 근무일, 진행중인 상담의 남은 날짜, 현재까지의 이익
    global maxV
    if n == N:
        if maxV < bene:
            maxV = bene
    else:
        if r < 1 and T[n] + n <= N:
            f(n + 1, N, T[n] - 1, bene + P[n])
        f(n + 1, N, r - 1, bene)


N = int(input())
T, P = [0]*N, [0]*N
for i in range(N):
    t, p = map(int, input().split())
    T[i] = t
    P[i] = p

maxV = 0
f(0, N, 0, 0)
print(maxV)
```
