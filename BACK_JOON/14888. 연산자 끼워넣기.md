                result *= A[j]
            elif show[j-1] == 3:
                if result < 0:
                    result *= -1

                    result *= -1
                else:
                    result //= A[j]

N = int(input())
A = list(map(int, input().split()))

temp = [0]*(N-1)
visited = [0]*(N-1)
i, idx = 0, 0
while idx < N-1:
    if B[i] > 0:
        temp[idx] = i

        idx += 1
    elif B[i] == 0:
