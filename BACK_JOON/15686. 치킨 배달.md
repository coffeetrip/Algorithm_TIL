def f2(M):
    s=0 # 도시치킨거리(각 집에서 치킨집까지 최소거리의 합)
    for i in range(len(home)): # 각 집에 대해
        minH = 10000000
        for j in range(len(comb)): #  선택한 각 치킨집까지
            dis = abs(home[i][0]-bbq[comb[j]][0]) + abs(home[i][1]-bbq[comb[j]][1]) # 거리계산
            if minH>dis: # 어떤 집에서 치킨집까지의 최소거리(치킨거리)
                minH = dis
        s += minH
    return s

def f(n, s, k, M): # n 순서, s 고를 수 있는 치킨집 시작번호, k 전체 치킨집 수, M 고를 치킨집 수
    global minV
    if n==M: # M개의 치킨집을 고르면
        r = f2(M)    # 도시 치킨거리 계산
        if minV>r:
            minV = r
    else:
        for i in range(s, k-(M-n)+1):
            comb[n] = i # i번 치킨집을 선택
            f(n+1, i+1, k, M)

            bbq.append((i, j))
        elif arr[i][j]==1:

comb = [0]*M
f(0, 0, k, M)

def comb(p, n, k):
    global chicken_list
    if k == 0:
        city_chicken(p)
    elif n < k:
        return
    else:
        p[k - 1] = chicken_list[n - 1]
        comb(p, n - 1, k - 1)
        comb(p, n - 1, k)


def city_chicken(p):
    global home_list, minV
    total = 0
    for home in home_list:
        temp = 1000000
        for chicken in p:
            distance = abs(home[0] - chicken[0]) + abs(home[1] - chicken[1])
            if temp > distance:
                temp = distance
        total += temp
    if minV > total:
        minV = total


N, M = map(int, input().split())
arr = [list(map(int, input().split())) for _ in range(N)]
# 집과 치킨집의 위치 찾기
home_list = []
chicken_list = []
for r in range(N):
    for c in range(N):
        if arr[r][c] == 1:
            home_list.append((r, c))
        elif arr[r][c] == 2:
            chicken_list.append((r, c))
# 최대 M 개의 치킨집 정하기
minV = 1000000
p = [0] * M
comb(p, len(chicken_list), M)
