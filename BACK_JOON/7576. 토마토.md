```python
from collections import deque


def bfs(tomato):
    global M, N
    visited = [[0] * M for _ in range(N)]
    queue = deque()
    tomato_cnt, day = 0, 0
    for i in range(N):
        for j in range(M):
            if tomato[i][j] == 1:
                visited[i][j] = 1
                tomato_cnt += 1
                queue.append((i, j))
            elif tomato[i][j] == -1:
                visited[i][j] = -1
                tomato_cnt += 1

    while queue:
        si, sj = queue.popleft()
        di = [0, 1, 0, -1]
        dj = [1, 0, -1, 0]
        for k in range(4):
            ni = si + di[k]
            nj = sj + dj[k]
            if 0 <= ni < N and 0 <= nj < M and visited[ni][nj] == 0:
                if tomato[ni][nj] == 0:
                    visited[ni][nj] = visited[si][sj] + 1
                    tomato_cnt += 1
                    queue.append((ni, nj))
                    if day < visited[si][sj]:
                        day = visited[si][sj]

    if M * N - tomato_cnt > 0:
        return -1
    # print(visited)
    return day


M, N = map(int, input().split())
tomato = [list(map(int, input().split())) for _ in range(N)]
print(bfs(tomato))
```
