```PYTHON
def gcd(x, y):
    while y != 0:
        x, y = y, x % y
    return x

t = int(input())
for _ in range(t):
    n, *num = map(int, input().split())

    s = 0
    for i in range(0, n-1):
        for j in range(i+1, n):
            s += gcd(num[i], num[j])
    print(s)
```
