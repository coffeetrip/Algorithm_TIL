# 1012. 유기농 배추

```python
def bfs(si, sj):
    global visited
    front, rear = -1, 0
    queue = [0]*(N*M)
    queue[rear] = [si, sj]
    visited[si][sj] = 1
    while front != rear:
        di = [0, 1, 0, -1]
        dj = [1, 0, -1, 0]
        front += 1
        i, j = queue[front]
        for k in range(4):
            ni = i + di[k]
            nj = j + dj[k]
            if 0 <= ni < N and 0 <= nj < M and visited[ni][nj] == 0 and farm[ni][nj] == 1:
                visited[ni][nj] = 1
                farm[ni][nj] = 0
                rear += 1
                queue[rear] = [ni, nj]
    return 1

T = int(input())
for tc in range(1, T+1):
    M, N, K = map(int, input().split())  # 가로, 세로, 배추개수
    farm = [[0]*M for _ in range(N)]
    visited = [[0]*M for _ in range(N)]
    for _ in range(K):
        y, x = map(int, input().split())
        farm[x][y] = 1

    cnt = 0
    for i in range(N):
        for j in range(M):
            if farm[i][j] == 1:
                cnt += bfs(i, j)
    print(cnt)
```

```python
    while queue:

        dj = [1, 0, -1, 0]
        i, j = queue.pop(0)
```
