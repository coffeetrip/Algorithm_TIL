def bfs():
    visited = [[0] * M for _ in range(N)]
    q = [(0, 0)]
    visited[0][0] = 1
    while q:
        i, j = q.pop(0)
        for di, dj in (-1, 0), (0, 1), (1, 0), (0, -1):
            ni, nj = i + di, j + dj
            if 0 <= ni < N and 0 <= nj < M and visited[ni][nj] == 0:
                if arr[ni][nj] == 0:
                    visited[ni][nj] = 1
                    q.append((ni, nj))
                else:
                    arr[ni][nj] += 1


def melt():
    global cheese
    cnt, result = 0, 0
    for i in range(N):
        for j in range(M):
            if arr[i][j] >= 2:
                arr[i][j] = 0
                cnt += 1
                result = True
    if cnt:
        cheese = cnt
    return result

N, M = map(int, input().split())  # 세로, 가로
arr = [list(map(int, input().split())) for _ in range(N)]

time, cheese = 0, 0
while True:
    bfs()
    if melt():
        time += 1
    else:
        break

print(time)
print(cheese)
