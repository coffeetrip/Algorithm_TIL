### DFS - 재귀

```PYTHON
def f2(i, j, N):
    global homecnt
    v[i][j] = 1  # 방문표시
    homecnt += 1
    # 유효한 인덱스인지 먼저 검사하기, 독립적으로 움직이기 때문에 각각의 if문을 사용
    if 0 <= j + 1 < N and arr[i][j + 1] == '1' and v[i][j + 1] == 0:
        f2(i, j + 1, N)
    if 0 <= i + 1 < N and arr[i + 1][j] == '1' and v[i + 1][j] == 0:
        f2(i + 1, j, N)
    if 0 <= j - 1 < N and arr[i][j - 1] == '1' and v[i][j - 1] == 0:
        f2(i, j - 1, N)
    if 0 <= i - 1 < N and arr[i - 1][j] == '1' and v[i - 1][j] == 0:
        f2(i - 1, j, N)
```

```PYTHON
def f(i, j, N):
    global homecnt
    v[i][j] = 1  # 방문표시
    homecnt += 1
    for di, dj in (-1, 0), (0, 1), (1, 0), (0, -1):
            ni, nj = i + di, j + dj
        if 0 <= ni < N and 0 <= nj < N:
            if arr[ni][nj] == '1' and v[ni][nj] == 0:
                f(ni, nj, N)
```

```PYTHON
N = int(input())
arr = [list(input()) for _ in range(N)]
v = [[0]*N for _ in range(N)]  # 단지에 이미 포함된 위치 표시
home = []

cnt = 0
for i in range(N):
    for j in range(N):
        if arr[i][j] == '1' and v[i][j] == 0:  # 단지에 속하지 않은 건물을 찾으면
            homecnt = 0  # 단지 내부 건물 수 카운트
            f2(i, j, N)
            cnt += 1  # 단지의 수
            home.append(homecnt)

home.sort()
print(cnt)
for x in home:
    print(x)
```

```PYTHON
def bfs(i, j):
    q = []
    q.append((i, j))
    visited[i][j] = 1
    cnt = 0
    while q:
        si, sj = q.pop(0)
        cnt += 1
  		for di, dj in [(1, 0), (-1, 0), (0, 1), (0, -1)]:
            ni, nj = si + di, sj + dj
            if 0 <= ni < N and 0 <= nj < N and zido[ni][nj] == 1 and visited[ni][nj] == 0:
                visited[ni][nj] = visited[si][sj] + 1
                q.append((ni, nj))
    return cnt
```

```PYTHON
N = int(input())
zido = [list(map(int, input())) for _ in range(N)]
visited = [[0]*N for _ in range(N)]

totalcnt = 0
home = []
for i in range(N):
    for j in range(N):
        if zido[i][j] == 1 and visited[i][j] == 0:
            totalcnt += 1
```
