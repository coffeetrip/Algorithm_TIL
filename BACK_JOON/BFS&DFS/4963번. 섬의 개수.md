# 4963번. 섬의 개수

```python
def bfs(si, sj):
    global visited, cnt
    queue = [0]*(W*H*2)
    front, rear = -1, 0
    queue[rear] = [si, sj]
    visited[si][sj] = 1

    while front != rear:
        front += 1
        i, j = queue[front]
        di = [0, 1, 1, 1, 0, -1, -1, -1]
        dj = [-1, -1, 0, 1, 1, 1, 0, -1]
        for k in range(8):
            ni = i + di[k]
            nj = j + dj[k]
            if 0 <= ni < H and 0 <= nj < W and visited[ni][nj] == 0 and zido[ni][nj] == 1:
                visited[ni][nj] = 1
                zido[ni][nj] = 0
                rear += 1
                queue[rear] = [ni, nj]
    return 1

while True:
    W, H = map(int, input().split())
    if W == 0 and H == 0:
        break
    zido = [list(map(int, input().split())) for _ in range(H)]
    visited = [[0]*W for _ in range(H)]

    cnt = 0
    for i in range(H):
        for j in range(W):
            if zido[i][j] == 1:
                cnt += bfs(i, j)
    print(cnt)
```
