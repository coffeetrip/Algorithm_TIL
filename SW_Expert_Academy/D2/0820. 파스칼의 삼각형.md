# 0820. 파스칼의 삼각형

```
for i :0 -> n-1  # row 행 번호
	for j :0 -> i  # col 열 번호
		if j==0 or i==j  # 가장자리
			arr[i][j] = 1
		else
			arr[i][j] = arr[i-1][j-1] + arr[i-1][j]
```

```python
for tc in range(int(input())):
    n = int(input())
    arr = [[0] * n for _ in range(n)]

    result = []
    print(f'#{tc+1}')
    for i in range(n):
        for j in range(i+1):
            if j == 0 or i == j:
                arr[i][j] = 1
            else:
                arr[i][j] = arr[i-1][j-1] + arr[i-1][j]
            print(arr[i][j], end=" ")
        print()
```

```python
t = int(input())
for tc in range(t):
    n = int(input())
    arr = [[] for y in range(n)]    # [[], [], [], []]

    print(f'#{tc+1}')
    for i in range(n):   # row 행 번호
        for j in range(i+1):   # col 열 번호
            if j == 0 or i == j:   # 가장자리
                arr[i].append(1)
            else:
                arr[i].append(arr[i-1][j-1] + arr[i-1][j])
        result = ' '.join(list(map(str, arr[i])))
        print(f'{result}')
```

```python
for T in range(int(input())):
    N = int(input())
    arr = [[0] * N for _ in range(N)]
    result = ''
    for i in range(N):
        for j in range(i+1):
            if j == 0 or i == j:
                arr[i][j] = 1
            else:
                arr[i][j] = arr[i-1][j-1] + arr[i-1][j]
            result += str(arr[i][j]) + ' '
        result += '\n'
    print(f'#{T+1}\n{result}', end='')
```
