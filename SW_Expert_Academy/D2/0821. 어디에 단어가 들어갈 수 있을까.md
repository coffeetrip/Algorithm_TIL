def col(i, j):
    global N, K
    global puzzle
    length = 1
    dj = j - 1
    for t in range(1, N):
        nj = j + 1*t
        if nj>=0 and nj<N:
            if j == 0:
                if puzzle[i][nj] == 0:
                    break
                else:
                    length += 1
            else:
                if puzzle[i][dj] != 1:
                    if puzzle[i][nj] == 0:
                        break
                    else:
                        length += 1
    if length == K:
        return 1
    return 0

def row(i, j):
    global N, K
    global puzzle
    length = 1
    di = i - 1
    for t in range(1, N):
        ni = i + 1*t
        if ni>=0 and ni<N:
            if i == 0:
                if puzzle[ni][j] == 0:
                    break
                else:
                    length += 1
            else:
                if puzzle[di][j] != 1:
                    if puzzle[ni][j] == 0:
                        break
                    else:
                        length += 1
    if length == K:
        return 1
    return 0

for tc in range(int(input())):
    n, k = map(int, input().split())

    puzzle = [list(map(int, input().split())) for _ in range(n)]
    word = [1]*k
    reverse_puzzle = list(map(list, zip(*puzzle)))

    count = 0
    for i in range(n):
        for j in range(n-k+1):
            if j == 0 and word + [0] == puzzle[i][j:j+k+1]:
                count += 1
            elif j == n-k and [0] + word == puzzle[i][j-1:j+k]:
                count += 1
            elif [0] + word + [0] == puzzle[i][j-1:j+k+1]:
                count += 1
            if j == 0 and word + [0] == reverse_puzzle[i][j:j+k+1]:
                count += 1
            elif j == n-k and [0] + word == reverse_puzzle[i][j-1:j+k]:
                count += 1
            elif [0] + word + [0] == reverse_puzzle[i][j-1:j+k+1]:
                count += 1
    print(f'#{tc+1} {count}')

    n, k = map(int, input().split())
    nums = [list(map(int, input().split())) for x in range(n)]
    #가로 탐색위치
    cnt = 0
    for i in range(n): # n * n이니까 반복문이 하나만 있어도 됨
        row_idx = 0 #반복될때 마다 값을 초기화해야되기 때문에
        col_idx = 0
        length = 0
        while row_idx < n: # 다 돌면 끝나려고 넣는거고
            while row_idx <n and nums[row_idx][i] ==0: #앞에 row_idx가 없으면 인덱스범위를 벗언서 에러
                row_idx += 1
            length = 0 # 0을 만났을때 값을 초기화시켜줘야함
            while row_idx < n and nums[row_idx][i] ==1:
                length += 1
                row_idx +=1
            if length== k:
                cnt += 1
    
        while col_idx < n:
            while col_idx <n and nums[i][col_idx]==0:
                col_idx += 1
            length = 0
            while col_idx < n and nums[i][col_idx]==1:
                length +=1
                col_idx +=1
            if length == k:
                cnt +=1
    print('#{} {}'.format(tc, cnt))
