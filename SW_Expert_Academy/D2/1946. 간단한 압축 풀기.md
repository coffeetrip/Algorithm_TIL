# 1946. 간단한 압축 풀기

```python
T = int(input())
for tc in range(1, T+1):
    N = int(input())
    s = ''
    for nc in range(N):
        C, K = input().split()
        s += C*int(K)

    print('#{}' .format(tc))
    i = 0
    k = len(s)
    while k > 10:
        print(s[i:i+10])
        i += 10
        k -= 10
    print(s[i:i+k])
```

```python
T = int(input())
for tc in range(1, T+1):
    code = ""
    n = int(input())
    for i in range(n):
        char, cnt = input().split()
        code += char*int(cnt)

    for j in range((len(code)//10)+1):
        print(code[j*10:j*10+10])
```
