```python
def line(sudoku_list):
    for i in range(9):
        line = [0] * 9
        for j in sudoku_list[i]:
            line[j - 1] = 1
        if sum(line) != 9:
            return 0
    return 1

def grid(sudoku_list):
    for i in range(0, 9, 3):
        grid = [0]*9
        for k in range(3):
            for c in range(3):
                grid[sudoku_list[i+k][i+c]-1] = 1
        if sum(grid) != 9:
            return 0
    return 1

T = int(input())
for tc in range(1, T+1):
    sudoku = [list(map(int, input().split())) for _ in range(9)]

    sudoku_90 = []
    for i in range(9):
        sudoku_90_row = []
        for j in range(9):
            sudoku_90_row.append(sudoku[j][i])
        sudoku_90.append(sudoku_90_row)

    result = 0
    if line(sudoku) == 1 and line(sudoku_90) == 1 and grid(sudoku) == 1:
        result = 1
    else:
        result = 0

    print('#{} {}' .format(tc, result))
```
