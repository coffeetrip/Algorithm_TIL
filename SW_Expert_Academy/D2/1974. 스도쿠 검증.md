```python
T = int(input())
for tc in range(1, T+1):
    arr = [list(map(int, input().split())) for _ in range(9)]
 
    result = 1
    for i in range(9):
        row_list = [0]*10
        col_list = [0]*10
        for j in range(9):
            row_list[arr[i][j]] = 1
            col_list[arr[j][i]] = 1
        if sum(row_list) != 9:
            result = 0
        if sum(col_list) !=9:
            result = 0
 
    for i in range(0, 9, 3):
        x3_list = [0]*10
        for j in range(3):
            for k in range(3):
                x3_list[arr[i+j][i+k]] = 1
        if sum(x3_list) != 9:
            result = 0
    print('#{} {}' .format(tc, result))
```

```python
for tc in range(1, int(input())+1):
    arr = [list(map(int, input().split())) for _ in range(9)]

    result = 1
    
    for i in range(9):
        sum_row = 0
        sum_col = 0
        for j in range(9):
            sum_row += arr[j][i]
            sum_col += arr[i][j]
        if sum_row != sum(range(1, 10)) or sum_col != sum(range(1, 10)):
            result = 0

    for i in range(0, 9, 3):
        for j in range(0, 9, 3):
            x_sum = 0
            for k in range(3):
                for z in range(3):
                    x_sum += arr[i+k][j+z]
            if x_sum != sum(range(10)):
                result = 0

    print(f'#{tc} {result}')
```
