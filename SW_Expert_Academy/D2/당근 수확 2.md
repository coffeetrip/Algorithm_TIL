###  당근 수확 2

```python

T = int(input())
for tc in range(1, T+1):
    N, M = map(int, input().split())
    num = list(map(int, input().split()))

    move = 0
    idx = -1
    while idx < N-1:
        idx += 1
        if idx > 0:
            move += 1
        if num[idx] >= M:
            num[idx] -= M
            move += idx
            idx = -1
        else:
            if idx != N-1:
                num[idx+1] += num[idx]
                num[idx] = 0
            else:
                move += N

        print(num, move, idx)
```
