def check_palindrome_one_line(line, find, n):
    for i in range(0, n):
        f_char = line[i]

        for j in range(i + 1, n):
            b_char = line[j]

            if f_char == b_char:
                word = line[i:j + 1]
                if len(word) > find and word == word[::-1]:
                    find = len(word)

    return find


for t in range(10):
