# 1860. 진기의 최고급 붕어빵

```python
T = int(input())
for tc in range(1, T+1):
    N, M, K = map(int, input().split())
    s = list(map(int, input().split()))

    max_s = max(s)
    new = [0]*(max_s+1)
    for q in range(len(s)):
        new[s[q]] -= 1

    for j in range(M, len(new), M):
        new[j] += K
    for k in range(1, len(new)):
        if (new[k-1] != 0 and new[k] == 0) or new[k-1] != 0 and new[k] != 0:
            new[k] += new[k-1]

    result = 0
    for x in new:
        if x < 0:
            result = 1

    if result == 0:
        print('#{} Possible' .format(tc))
    else:
        print('#{} Impossible'.format(tc))
```
