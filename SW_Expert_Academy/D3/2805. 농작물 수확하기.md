```
center = n//2
s = 0
for i:0 -> n-1
	if i <= center
		for j:0 -> n-1
			if j >= center-i and j<=center +i
				s += arr[i][j]
	else
		for j:0 -> n-1

				s += arr[i][j]
```

```python
T = int(input())
for tc in range(1, T+1):
    N = int(input())
    farm = [list(map(int, input())) for _ in range(N)]

    s = 0
    k = 0
    while k < N//2:
        for i in range(N//2):
            s += sum(farm[i][N//2-k:N//2+k+1])
            k += 1

    s += sum(farm[N//2])

    k2 = N//2 - 1
    while k2 >= 0:
        for i in range(N//2+1, N):
            s += sum(farm[i][N//2-k2:N//2+k2+1])
            k2 -= 1
    print('#{} {}' .format(tc, s))
```

```python
# 수정필요
T = int(input())
for tc in range(1, T+1):
    N = int(input())
    farm = [list(map(int, input())) for _ in range(N)]

    s = 0
    k = 0
    while k <= N//2:
        if k != N//2:
            for i in range(N//2+1):
                s += sum(farm[i][N//2-k:N//2+k+1])
                s += sum(farm[-(i+1)][N//2-k:N//2+k+1])
                k += 1
        else:
            s += sum(farm[N//2])
    print('#{} {}' .format(tc, s))
```
