# 4047. 영준이의 카드 카운팅(tn)

```python
T = int(input())
for tc in range(1, T+1):
    card = input()
    S = [0]*14
    D = [0]*14
    H = [0]*14
    C = [0]*14

    for i in range(len(card)):
        if card[i] == 'S':
            S[int(card[i+1:i+3])] += 1
        elif card[i] == 'D':
            D[int(card[i+1:i+3])] += 1
        elif card[i] == 'H':
            H[int(card[i+1:i+3])] += 1
        elif card[i] == 'C':
            C[int(card[i+1:i+3])] += 1

    for i in S:
        if i != 0 or i != 1:
            result = 'ERROR'
        else:
            result1 = 13-sum(S)
    for i in D:
        if i != 0 or i != 1:
            result = 'ERROR'
        else:
            result2 = 13 - sum(D)
    for i in H:
        if i != 0 or i != 1:
            result = 'ERROR'
        else:
            result3 = 13-sum(H)
    for i in C:
        if i != 0 or i != 1:
            result = 'ERROR'
        else:
            result4 = 13-sum(C)
    if result == 'ERROR':
        print('#{} ERROR' .format(tc))
    else:
        print('#{} {} {} {} {}'.format(tc, result1, result2, result3, result4))
```
