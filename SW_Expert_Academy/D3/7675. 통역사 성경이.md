### 7675. 통역사 성경이

```python
T = int(input())
for tc in range(1, T+1):
    N = int(input())
    sentence = input()

    div = []
    s = ''
    for i in range(len(sentence)):
        if sentence[i] == '.' or sentence[i] == '?' or sentence[i] == '!':
            div.append(s)
            s = ''
        else:
            s += sentence[i]

    cnt_list = []
    for i in range(N):
        a = div[i].split()
        cnt = 0
        for j in range(len(a)):
            aj1 = [0]*(len(a[j])-1)
            for k in range(1, len(a[j])):
                if ord(a[j][k]) in range(97, 123):
                    aj1[k-1] = 1
            if ord(a[j][0]) in range(65,91) and sum(aj1) == len(a[j])-1:
                cnt += 1
        cnt_list.append(cnt)
    print('#{}' .format(tc), end=" ")
    print(*cnt_list)
```
