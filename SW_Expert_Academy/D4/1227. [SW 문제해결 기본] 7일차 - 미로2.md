### 1227. [S/W 문제해결 기본] 7일차 - 미로2

```python
def find(i, j):
    global visited
    q = [[i, j]]
    visited[i][j] = 1
    while len(q) != 0:
        n = q.pop(0)
        di = [0, 1, 0, -1]
        dj = [1, 0, -1, 0]
        for k in range(4):
            ni = n[0] + di[k]
            nj = n[1] + dj[k]
            if 0 <= ni < 100 and 0 <= nj < 100 and visited[ni][nj] == 0 and miro[ni][nj] != 1:
                if miro[ni][nj] == 3:
                    return 1
                else:
                    q.append([ni, nj])
                    visited[ni][nj] = 1
    return 0


for _ in range(10):
    tc = int(input())
    miro = [list(map(int, input())) for _ in range(100)]
    visited = [[0]*100 for _ in range(100)]

    result = find(1, 1)

    print('#{} {}' .format(tc, result))
```
