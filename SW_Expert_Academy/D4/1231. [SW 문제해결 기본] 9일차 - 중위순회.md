def order(n, N):
    global tree
    if n <= N:
        order(2*n, N)
        print(tree[n], end="")
        order(2*n+1, N)

for i in range(1, 11):
    N = int(input())
    tree = [0]*(N+1)
    for nc in range(N):
        num = list(input().split())
        tree[int(num[0])] = num[1]

    print('#{}' .format(i), end=" ")
    order(1, N)
    print()

def inorder(n):  # 중위 순회
    global word
    if n > 0:
        inorder(ch1[n])
        word.append(alp[n])
        inorder(ch2[n])
 
 
for tc in range(1, 11):
    V = int(input())
 
    par = [0] * (V + 1)
    alp = [""] * (V + 1)
    ch1 = [0] * (V + 1)
    ch2 = [0] * (V + 1)
 
    word = []
    for i in range(1, V + 1):
        tree = list(map(str, input().split()))
        for j in range(len(tree)):
            if j == 0:
                par[i] = int(tree[j])
            elif j == 1:
                alp[i] = tree[j]
            elif j == 2:
                ch1[i] = int(tree[j])
            else:
                ch2[i] = int(tree[j])
 
    inorder(1)
    print("#{}".format(tc), end=" ")
    for a in range(len(word) - 1):
        print(word[a], end="")
    print(word[-1])
