### [S/W 문제해결 기본] 7일차 - 미로1

```python
def find(i, j):
    global miro

    if miro[i][j] == 3:
        return 1
    else:
        miro[i][j] = 1
        d = [(0, 1), (0, -1), (1, 0), (-1, 0)]
        for dx, dy in d:
            ni = i + dx
            nj = j + dy
            if 0 <= ni < 16 and 0 <= nj < 16 and miro[ni][nj] != 1:
                if find(ni, nj) == 1:
                    return 1
    return 0


for tc in range(1, 11):
    test_case = int(input())
    miro = [list(map(int, input())) for _ in range(16)]

    print('#{} {}' .format(test_case, find(1, 1)))
```

```python
def explore(i, j):
    global miro
    global visited

    if miro[i][j] == '3':
        return 1
    else:
        visited[i][j] = 1
        d = [(0, 1), (0, -1), (1, 0), (-1, 0)]

        for dx, dy in d:
            ni = i + dx
            nj = j + dy
            if 0 <= ni < 16 and 0 <= nj < 16 and visited[ni][nj] == 0:
                if miro[ni][nj] != '1':
                    if explore(ni, nj) == 1:
                        return 1
    return 0
```

```python
def find(i, j):
    global miro
    global endIdx, endJdx
    di = [0, 1, 0, -1]
    dj = [1, 0, -1, 0]
    miro[i][j] = 1

    if (i, j) == (endIdx, endJdx):
        return 1
    else:
        for k in range(4):
            ni = i + di[k]
            nj = j + dj[k]
            if 0 <= ni < 16 and 0 <= nj < 16 and miro[ni][nj] != 1:
                if find(ni, nj) == 1:
                    return 1
    return 0


for tc in range(1, 11):
    test_case = int(input())
    miro = [list(map(int, input())) for _ in range(16)]

    endIdx = 0
    endJdx = 0
    for i in range(16):
        for j in range(16):
            if miro[i][j] == 3:
                endIdx = i
                endJdx = j

    print('#{} {}' .format(test_case, find(1, 1)))
```
