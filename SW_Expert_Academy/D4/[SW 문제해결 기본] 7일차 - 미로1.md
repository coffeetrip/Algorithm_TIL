### [S/W 문제해결 기본] 7일차 - 미로1

```python
def find(i, j):
    global miro
    global result
    global endI, endJ
    di = [0, 1, 0, -1]
    dj = [1, 0, -1, 0]
    miro[i][j] = 1

    if (i, j) == (endI, endJ):
        result = 1
    else:
        for k in range(4):
            ni = i + di[k]
            nj = j + dj[k]
            if 0<=ni<16 and 0<=nj<16:
                if miro[ni][nj] != 1:
                    find(ni, nj)


for _ in range(1, 11):
    tc = int(input())
    miro = [list(map(int, input())) for _ in range(16)]

    endI = 0
    endJ = 0
    result = 0
    for i in range(16):
        for j in range(16):
            if miro[i][j] == 3:
                endI = i
                endJ = j

    find(1, 1)
    print('#{} {}' .format(tc, result))
```
