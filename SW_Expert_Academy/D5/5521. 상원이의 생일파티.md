### 5521. 상원이의 생일파티

```python
def f(n):
    global card, cnt
    queue = []
    card[n] = 0

    if not p[1]:
        return 0
    queue.append(1)

    while queue:
        x = queue.pop(0)
        for i in p[x]:
            if i != 1 and card[i] == 0:
                card[i] = card[x] + 1
                queue.append(i)
    for i in card:
        if 1<= i <= 2:
            cnt += 1
    return cnt

T = int(input())
for tc in range(1, T + 1):
    N, M = map(int, input().split())

    p = {i: [] for i in range(1, N + 1)}
    for mc in range(M):
        a, b = map(int, input().split())
        p[a].append(b)
        p[b].append(a)

    card = [0 for i in range(N + 1)]
    cnt = 0
    print('#{} {}' .format(tc, f(1)))
```
