### 반복문(스택 이용) - DFS

```python
# 반복문(스택이용)
def find(sRow, sCol):
    dRow = [0, 1, 0, -1]
    dCol = [1, 0, -1, 0]
    s = []
    s.append([sRow,sCol])    # 입구로 이동
    maze[sRow][sCol] = 1     # 방문 표시
    while len(s) != 0:
        n = s.pop()          # 이동할 칸 좌표를 꺼내고(이동할 수 없을때 뒤로 이동)
        for i in range(4):    # 주변 좌표 계산
            nRow = n[0] + dRow[i]
            nCol = n[1] + dCol[i]
            if nRow >=0 and nRow<N and nCol>=0 and nCol <N:  # 미로 내부인지 확인
                if maze[nRow][nCol] == '3':     # 목적지인 경우 1반환
                    return 1
                elif maze[nRow][nCol] == '0':    # 갈 수 있는 곳 저장
                    s.append([nRow, nCol])
                    maze[n[0]][n[1]] = 1
    return 0    # 출구에 가지 못하고 모든칸 방문

T = int(input())
for tc in range(1, T+1):
    N = int(input())
    maze = [list(input()) for _ in range(N)]

    startI = 0
    startJ = 0
    for i in range(N):
        for j in range(N):
            if maze[i][j] == '2':
                startI = i
                startJ = j
    print('#{} {}' .format(tc, find(startI, startJ)))
```

# 재귀를 이용한 미로찾기
def f(i, j):
    di = [0, 1, 0, -1]
    dj = [1, 0, -1, 0]
    global maze
    global N
    # if maze[i][j] == '1':   # 벽이면
    #     return 0
    if maze[i][j] == '3':   # 목적지면
        return 1
    else:
        maze[i][j] = '1'   # 방문 표시, 벽으로 바꿈
        # 이동할 좌표 생성
        for k in range(4):
            ni = i + di[k]
            nj = j + dj[k]
            if ni>=0 and ni<N and nj>=0 and nj<N:
                if maze[ni][nj] != '1':  # 벽이 아니면 방문 (0,3)
                    if f(ni, nj) == 1:  # 진행방향에서 목적지를 찾은 경우
                        return 1
        return 0  # 현재위치에서 갈 수 있는 방햐에서 목적지를 찾지 못함. 이전의 다른 방향 탐색


T = int(input())
for tc in range(1, T+1):

    # 시작점 찾기
    startI = 0

                startI = i

## 미로의 거리

### BFS 

```PYTHON
def bfs(i, j):
    global N
    global maze
    di = [0, 1, 0, -1]
    dj = [1, 0, -1, 0]

    queue = []  
    visited = [[0 for _ in range(N)] for _ in range(N)]
    queue.append([i, j])  # 시작점 인큐
    visited[i][j] = 1  # 시작점 방문표시

    while len(queue) != 0:  # 큐가 비어있지 않으면 반복
        n = queue.pop(0)  # 디큐
        i, j = n[0], n[1]
        if maze[i][j] == 3:  # visit()
            return visited[i][j] - 2
        for k in range(4):  # i, j 에 인접하고 방문하지 않은 칸을 인큐
            ni = i + di[k]
            nj = j + dj[k]
            if ni>=0 and ni<N and nj>=0 and nj<N:  
                if maze[ni][nj] != 1 and visited[ni][nj] == 0: 
                    queue.append([ni, nj])  # 인큐
                    visited[ni][nj] = visited[i][j] + 1  # 방문표시
    return 0

for tc in range(int(input())):
    N = int(input())
    maze = [list(map(int, input())) for _ in range(N)]

    startI = 0
    startJ = 0
    for i in range(N):
        for j in range(N):
            if maze[i][j] == 2:
                startI = i
                startJ = j
    print(f'#{tc+1} {bfs(startI, startJ)}')
```

## 경로

```PYTHON
def f(i, j, e, N):  # 좌표, 지나온 칸 수, 미로의 크기
    global minV
    di = [0, 1, 0, -1]
    dj = [1, 0, -1, 0]
    if miro[i][j] == 3:
        if minV > e:
            minV = e
    else:
        miro[i][j] = 1  # 현재칸에 방문 표시(진행 방향에서 다시 들어오는 것 방지)
        for k in range(4):  # 이동 가능한 칸으로 이동
            ni = i + di[k]
            nj = j + dj[k]
            if ni>=0 and ni<N and nj>=0 and nj<N:
                if miro[ni][nj] != 1:  # 벽이 아니면(갈 수 있는 칸)
                    f(ni, nj, e+1, N)
        miro[i][j] = 0  # 새로운 경로로 현재칸에 진입허용


T = int(input())
for tc in range(1, T+1):
    N = int(input())
    miro = [list(map(int, input())) for _ in range(N)]

    startI = 0
    startJ = 0
    for i in range(N):
        if 2 in miro[i]:
            startI = i
            startJ = miro[i].index(2)
    minV = 1000000000000
    f(startI, startJ, 0, N)
    if minV == 1000000000000:
        minV = 1
    print('#{} {}' .format(tc, minV-1))
```
