```python
def positive_sum(numbers):
    result = 0
    for number in numbers:
        if number > 0:
            result += number
    return result


# 실행 결과를 확인하기 위한 코드입니다. 수정하지 마시오.

    print(positive_sum([-1, -2, -3, -4])) #=> 0
```

```python
def calc(equation):
    equation.strip('')
    idx_list = []
    value_list = []
    for idx, value in enumerate(equation):
        if value == '+' or value == '-':
            idx_list.append(idx)    #[0,3,6,11]
            value_list.append(value)   # [- + - +]

    equation_list=[]
    for i in range(len(idx_list)):
        if i != (len(idx_list)-1):
            equation_list.append(equation[idx_list[i]+1:idx_list[i+1]])
        else:
            equation_list.append(equation[idx_list[i]+1:])
    #print(equation_list)   # ['12', '12', '7979', '9191']

    result_list=[]
    if equation[0] == '+' or equation[0] == '-':
        for j in range(len(equation_list)):
            result_list.append(value_list[j])
            result_list.append(equation_list[j])
    #print(result_list)      # ['-', '12', '+', '12', '-', '7979', '+', '9191']

    final_list = []
    for k in range(int(len(result_list)/2)):
        final = result_list[2*k] + result_list[2*k+1]
        final_list.append(final)     # ['-12', '+12', '-7979', '+9191']
    
    result = 0
    for m in final_list:
        result += int(m)

    return result
```

```python
    res = 0
    for i in plus_split:
        if i == '':
            continue
        minus_split = i.split('-')
        if minus_split[0] != '':
            res += int(minus_split[0])
        if len(minus_split) == 2:
```

```python
# 파일명을 변경하지 마시오.
# 아래에 클래스 Point와 Circle을 선언하시오.
class Point:
    def __init__(self, x, y):
        self.x = x
        self.y = y

    def __str__(self):
        return f'Point:({self.x},{self.y})'

class Circle:
    def __init__(self, center, r):
        self.center = (center.x , center.y)
        self.r = r

    def get_area(self):
        return self.r * self.r * 3.14

    def get_perimeter(self):
        return 2 * self.r * 3.14

    def get_center(self):
        return self.center

    def __str__(self):
        return f'Circle:{self.center},r:{self.r}'



# 실행 결과를 확인하기 위한 코드입니다. 수정하지 마시오.
if __name__ == '__main__':
    p1 = Point(0, 0)
    c1 = Circle(p1, 3)
    print(c1.get_area())
    print(c1.get_perimeter())
    print(c1.get_center())
    print(c1)
    p2 = Point(4, 5)
    c2 = Circle(p2, 1)
    print(c2.get_area())
    print(c2.get_perimeter())
    print(c2.get_center())
    print(c2)
```
